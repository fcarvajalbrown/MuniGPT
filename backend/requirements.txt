# MuniGPT backend dependencies (Python 3.12)
#
#   pip install -r requirements.txt
#
# Local inference runs via the bundled official llama.cpp `llama-server` binary
# (backend/bin/), NOT an in-process Python binding — the prebuilt llama-cpp-python
# wheels require CPU instructions (AVX-512) that many target machines lack, while
# the official binary does runtime CPU dispatch and runs on any x86-64.
#
# Web / API layer
fastapi
uvicorn[standard]
httpx                 # local llama-server HTTP calls, BCN corpus fetcher
ddgs                  # /search (DuckDuckGo), unofficial free client, no API key
pydantic

# RAG / vector store
lancedb
pyarrow
tantivy               # BM-25 full-text index for LanceDB hybrid search

# Corpus ingestion
pypdf
pymupdf               # convert.py fallback extractor (better on some PDFs)

# System / licensing
psutil                # RAM-based model tier selection (FR-15)
cryptography          # Ed25519 offline license verification (FR-08)
